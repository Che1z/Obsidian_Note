Q: 輸出一個九九乘法表。（矩陣輸出形式）

for (int i = 1; i < 10; i += 3)
{
    for (int j = 1; j < 10; j++)
    {
        for (int k = 0; k < 3; k++)
        {
            Console.Write($"{i + k} * {j} : {(i + k) * j}\t");
        }
        Console.WriteLine();
    }

    Console.WriteLine();
}

Q: 輸出一個九九乘法表。（矩陣輸出形式 - While語法）

int i = 1;
int j = 1;
int k = 0;
while (i < 10)
{

    while (j < 10)
    {
        while (k < 3)
        {
            Console.Write($"{i + k} * {j} : {(i + k) * j}\t");
            k++;
        }
        Console.WriteLine();
        k = 0;
        j++;
    }
    j = 1;
    Console.WriteLine();
    i += 3;
}
